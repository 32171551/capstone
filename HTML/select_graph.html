그래프는 정점(Vertex)과 간선(edge)으로 이뤄진 비선형 자료구조를 의미합니다. 트리와는 다르게 그래프는 정점마다 간선이 있거나 없거나에 따른 루트 노드, 부모와 자식이라는 개념이 존재하지 않으며, 그래프는 노드와 엣지로 구성되어 있으며 객체 간의 관계를 표현할 수 있는 자료구조로 네트워크 모델이며 실생활에서 다양한 예를 표현할 수 있다. 지하철 노선도, 도로, 네비게이션의 최단경로 등 실생활에 활용할 수 있는 방법이 많습니다.

class Graph {
  // possible to be omitted
  // without contructor class will create an empty object
  constructor() {}
 
  // methods
  addNode(data) {
    let index = this.length();
    this[index.toString()] = {
      data: data,
      from: [],
      to: [],
    };
 
    return data;
  }
 
  indexOf(data) {
    if (!this.length()) {
      return;
    }
    let targetIndex = 0;
    while (this[targetIndex.toString()].data !== data) {
      targetIndex++;
      if (targetIndex === this.length()) {
        return;
      }
    }
 
    return targetIndex;
  }
 
  addEdge(data1, data2) {
    let from = this.indexOf(data1);
    let to = this.indexOf(data2);
    if (from !== undefined && to !== undefined) {
      this[from].to.push(to);
      this[to].from.push(from);
      return [data1, data2];
    }
    return;
  }
 
  edgeList(data) {
    let index = this.indexOf(data);
    let list = [];
 
    let target = this[index.toString()];
 
    for (let i of target.from) {
      let edgeInfo = {
        from: i,
        weight: index + i,
      };
      list.push(edgeInfo);
    }
 
    for (let i of target.to) {
      let edgeInfo = {
        to: i,
        weight: index + i,
      };
      list.push(edgeInfo);
    }
 
    return list;
  }
 
  removeNode(data) {
    let index = this.indexOf(data);
    let target = this[index.toString()];
 
    for (let i of target.from) {
      this[i.toString()].to = this[i.toString()].to.filter((v) => index !== v);
    }
 
    for (let i of target.to) {
      this[i.toString()].from = this[i.toString()].from.filter(
        (v) => index !== v
      );
    }
 
    delete this[index.toString()];
    return data;
  }
 
  clear() {
    if (this.isEmpty()) {
      return "nothing to clear!";
    } else {
      for (let key in this) {
        delete this[key];
      }
      return "clear done!";
    }
  }
 
  isEmpty() {
    return this.length() ? false : true;
  }
 
  length() {
    let leng = 0;
    for (let key in this) {
      if (key) {
        leng++;
      } else {
        return 0;
      }
    }
 
    return leng;
  }
}
